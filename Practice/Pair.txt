public class Pair<T> {
    private T value1;      // Valor guardado no no
    private T value2;      // Valor guardado no no
    private Pair<T> next; // Referencia para o proximo no da lista
 
    // Construtor
    Pair(T v1, T v2, Pair<T> n) {
       value1 = v1;
       value2 = v2;
       next = n;
    }
 
    // Getters e Setters
    public T getValue1() { return value1; }
    public T getValue2() { return value2; }
    @Override
    public String toString() { return "("+value1+","+value2+")"; }
    public Pair<T> getNext()  { return next; }
    public void setValue1(T v) { value1=v; }
    public void setValue2(T v) { value2=v; }
    public void setNext(Pair<T> n) { next = n; }
 }
